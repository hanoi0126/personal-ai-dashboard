@app.get("/api/insights/weekly", response_model=WeeklyInsightResponse)

    current_user: User = Depends(get_current_user),
    db: Session = Depends(get_db)
):
    """é€±æ¬¡ã‚¤ãƒ³ã‚µã‚¤ãƒˆã‚’å–å¾—"""
